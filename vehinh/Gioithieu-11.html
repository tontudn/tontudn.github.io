["Tập con 0 phần tử",{}],
["Tập con 1 phần tử",[{a},{b},{c}]],
["Tập con 2 phần tử",[{a,b},{b,c},{a,c}]],
["Tập con 3 phần tử",{a,b,c}]
],tieude="Các tập con của tập A={a,b,c}",tilecot=[1,1.2]);

 

 

 

>

bangbieu[vebang]([
["Đạo hàm","Kết quả"],
["(sinx)'","cosx"],
["(cosx)'","-sinx"],
["(tanx)'",[1+tan(x)^2,1/cos(x)^2]],
["(cotx)'",[-1-cot(x)^2,-1/sin(x)^2]]
],tieude="Bảng đạo hàm các hàm lượng giác",tilecot=[1,2]);

 

 

 

>

ham:=[x^n,1/x,sin(x),cos(x),exp(x)]: # thay đổi các hàm số nếu bạn muốn
